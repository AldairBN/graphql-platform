using System.Collections.Generic;
using HotChocolate.Data.Projections;
using HotChocolate.Resolvers;
using HotChocolate.Types;

namespace HotChocolate.Data.SqlKata
{
    /// <inheritdoc/>
    public class SqlKataProjectionVisitorContext
        : ProjectionVisitorContext<SqlKataProjectionDefinition>
    {
        /// <inheritdoc/>
        public SqlKataProjectionVisitorContext(
            IResolverContext context,
            IOutputType initialType)
            : base(context, initialType, new SqlKataProjectionScope())
        {
        }

        /// <summary>
        /// The path from the root to the current position in the input object
        /// </summary>
        public Stack<string> Path { get; } = new();

        /// <summary>
        /// A list of already projected fields
        /// </summary>
        public Stack<SqlKataProjectionDefinition> Projections { get; } = new();
    }
}
